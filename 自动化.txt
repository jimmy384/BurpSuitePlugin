if 原生请求 报错
    后面3个都判断成X
    return
if 低权限数据包 成功
    if 原生数据包.status == 低权限数据包.status
        if 原生数据包.length == 低权限数据包.length
            if 原生数据包.url 在白名单中
                没越权,低权限数据包判断成Y
            else
                越权,低权限数据包判断成X
        else
            没越权,低权限数据包判断成Y
    else
        if 低权限数据包 是没权限的提示(401, 403, 或者包含特定提示权限的字符串)
            没越权,低权限数据包判断成Y
        else
            越权,低权限数据包判断成X(实际有可能是接口报错,反正提示越权后,测试人就会具体看具体报文)
if 未授权数据包 成功
    越权,低权限数据包判断成X
else
    if 低权限数据包 是没权限的提示(401, 403, 或者包含特定提示权限的字符串)
        没越权,低权限数据包判断成Y
    else
        越权,低权限数据包判断成X(实际有可能是接口报错,反正提示越权后,测试人就会具体看具体报文)
